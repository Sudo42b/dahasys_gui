       """인증서 PDF 생성"""
       if not PDF_AVAILABLE:
           print("PDF generation not available - reportlab not installed")
           return False
       try:
           doc = SimpleDocTemplate(output_path, pagesize=A4, ...)
           styles = self._setup_styles()
           elements = []
           self._build_header(elements, styles, result, code_info)
           self._build_data_table(elements, styles, result, axis_results, code_info)
           self._build_summary(elements, styles, result)
           self._build_footer(elements, styles, result)
           doc.build(elements)
           return True
       except Exception as e:

           traceback.print_exc()
           return False
