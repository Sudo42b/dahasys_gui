   def _safe_get(row, column_names, name, alt_name=None, default=None):
       """행 데이터에서 컬럼값 안전 조회"""
       idx = None
       if name in column_names:
           idx = column_names.index(name)
       elif alt_name and alt_name in column_names:
           idx = column_names.index(alt_name)
       if idx is not None and idx < len(row):

       return default
